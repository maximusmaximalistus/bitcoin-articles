	# lvscan 

	# mount /dev/vg/root /mnt

	# mount /dev/vg/home /mnt/home

	# mount /dev/sdX3 /mnt/boot

	# mount /dev/sdX2 /mnt/boot/efi

	# mount -o bind /proc /mnt/proc
	
	# mount -o bind /sys /mnt/sys
	
	# mount -o bind /dev /mnt/dev
